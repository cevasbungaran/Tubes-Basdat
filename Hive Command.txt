----Dimensional Table untuk amazon_books_reviews
CREATE TABLE product (
  product_id INT,
  product_parent INT,
  product_title STRING,
  product_category STRING,
  marketplace STRING,
  PRIMARY KEY (product_id)
);

CREATE TABLE review (
  review_id INT,
  review_headline STRING,
  review_body STRING,
  review_date DATE,
  PRIMARY KEY (review_id)
);
